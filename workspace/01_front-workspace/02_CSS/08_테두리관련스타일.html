테두리 과녈ㄴ 스타일(border~)

        선택자{
            /*테두리 모양(종류)*/
            border-[위치]-style: none | solid(실선) | dotted | dashed | doubled | groove | ridge | inset | outset;
        
            /*테두리 두께*/
            border-[위치]-width:값;

            /*테두리 색상*/
            border-color: 색상 정보;

            /*위의 항목 한번에 적용 */
            border: 두께 스타일 [색상](생략 가능: default:검은색);

            /*테두리 둥글게 표현 */
            border-[상하위치-좌우위치]-radius:값;

            /*테두리(박스) 그림자 효과 */
            box-shadow: 가로거리(x) 세로거리(y) 흐림정도 번짐정도 색상정보;
        }
